rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    match /stores/{storeId}/xCards/{cardId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    match /stores/{storeId}/{document=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
